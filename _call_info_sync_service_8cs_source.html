338  private CallDirectionType CalcCallDirection(CallInfoSync initialCdrLine, IGrouping<Guid, CallInfoSync> group)
339  {
340  /*
341  * Если в инициирующей записи поле "Вх. А-номер" имеет значение длинной более 5 символов,
342  * считать вызов входящим.
343  */
344  if (initialCdrLine.SrcIn.Length >= PSTN_NUMBER_MINIUM_LENGTH)
345  return CallDirectionType.IncomingCall;
346 
347  var realDstNumber = ExtractRecordWithOriginalDstNumber(initialCdrLine, group).DstIn;
348  //Если поле "Вх. Б-номер" содержит не только цифры, значит звонили на сервис,
349  //пытаемсе определить, вел ли сервис кудато еще
350  if (Regex.IsMatch(realDstNumber, @"\D"))
351  realDstNumber = group.Where(x => x.SrcIn == initialCdrLine.SrcIn
352  && Regex.IsMatch(x.DstIn, @"^\d+$"))
353  .FirstOrDefault()?.DstIn
354  ?? realDstNumber;//если сервис не вел на реальный номер, оставляем номер сервиса

356  //Иначе считать вызов внутренним.

497  private IEnumerable<GroupCallInfosSync> GetResultingCallInfoSync(IEnumerable<GroupCallInfosSync> groupsCallInfosSync)

499  var callInfoSyncWithGroupName = new List<GroupCallInfosSync>();
500  var loopCounter = 0;
501  foreach (var group in groupsCallInfosSync)
502  {

504  {
505  loopCounter++;

507  var callInfosResult = new List<CallInfoSync>();
508  // Если вызов входящий.
509  if (group.CallDirectionType == CallDirectionType.IncomingCall)
510  {
511  /*
512  * Находим все записи, для которых поле "Время соединения" содержит значение,
513  * Поле "Идентификатор адресата вызова" содержит значение, состоящее только из цифр,
514  * Значение поля "Идентификатор инициатора" равно значению поля "Идентификатор адресата вызова" инициирующей записи.

517  callInfoSync.ConnectTime.HasValue
518  && Regex.IsMatch(callInfoSync.TerminatorId ?? String.Empty, TERMINATOR_ID_PATTERN)
519  && callInfoSync.OriginatorId == initCallInfo.TerminatorId).ToList();
520  if (resultItemsIncomingCall.Any())
521  {
522  callInfosResult.AddRange(resultItemsIncomingCall);
523  }

525  * Если ни одной записи не найдено, то выбрать запись, для которой "Идентификатор адресата вызова" находится в диапазоне [2000-9999],
526  * "Идентификатор инициатора" равен "Идентификатор адресата вызова" инициирующей записи;
527  * Поле "Дата создания CDR" имеет максимальное значение (из всех записей, удовлетворяющих предыдущим условиям).
528  */
529  else
530  {
531  resultItemsIncomingCall = group.RawCallInfosSync.Where(x =>
532  x.OriginatorId == initCallInfo.TerminatorId

534  && x.DstIn != null

542  ?? resultCandidates.FirstOrDefault()
543  ?? group.InitCallInfoSync;//Ничего не нашли. Зничит инициируючая запись будет результатом

546 
547  }
548  }

559  // Находится в диапазоне [2000-9999], а "Идентификатор адресата вызова" содержит значение, состоящие только из цифр.
560  // И запись не является переадресацией

562  x.ConnectTime.HasValue
563  && (x.Dvo == null || !Regex.IsMatch(x.Dvo, DVO_CALL_FORWARDING_PATTERN))

576  * "Идентификатор инициатора" = "Идентификатор инициатора" инициирующей записи,
577  * "Идентификатор адресата вызова" находится в диапазоне [2000-9999],

579  */

581  x.OriginatorId == initCallInfo.OriginatorId
