readonly string Cordis.Core.Sip.Class5.Settings.Vtes.Repository.VteExternalNumberRepository._getByDomainIdQuery
private
Initial value:
= @"
select
pn.phone_number Id,
sv.domain VteId,
pl.prefix Prefix,
pl.name Number,
pn.lines ChannelsCount
from Configuration.getPhoneListBinding(getdate()) bind
join SIM.sim sim on sim.sim = bind.sim
join Voice.phone_list pl (nolock) on bind.phone_list = pl.phone_list
join SIM.phone_number sim_phone on sim_phone.sim = bind.sim
join Voice.phone_number pn (nolock) on pn.phone_number = sim_phone.phone_number
join SIM.phone_number2vte p2v on p2v.phone_number = bind.sim
join SIM.vte sv on sv.sim = p2v.vte
join Metrics.phone_number mpn on mpn.metric = sim.metric
join Voice.line_type lt on lt.line_type = mpn.line_type
where
lt.code = 'external_number'
and sv.domain = @domainId"
