127  private void BuildUplinkGateways(List<DomainSettings> domains, DataConnection data)
128  {
129  var groupSetIds = domains.Where(x => x.GroupSetId != null).Select(x => x.GroupSetId.Value);
130  //дальше можно все вообще переписать на один джоин с группами, но есть опасность что linq2db может превратить это в АД.
131  var sets2Groups = data.GetTable<GroupSet2Group>().Where(x => groupSetIds.Contains(x.GroupSetId));
132  var groupIds = sets2Groups.Select(x => x.GroupId);
133  //TODO это переделать бы на использование групп родительского домена. Сейчас гарантирована их загрузка на предыдущих этапах
134  var groupsWithSetId = data.GetTable<Group>()
135  .Where(x => groupIds.Contains(x.Id))
136  .Join(sets2Groups, x => x.Id, x => x.GroupId, (x, y) => new { g = x, s = y.GroupSetId }).ToList();
137 
138  foreach (var domain in domains)
139  {
140  domain.UpLinkGateway = domain.Gateways.FirstOrDefault(x => x.Id == domain.UpLinkGatewayId);
141  domain.GatewaySettings = domain.Gateways.FirstOrDefault(x => x.Id == domain.GatewaySettingsId);
142  domain.Gateways = domain.Gateways.Where(x => x.Id != domain.GatewaySettingsId).ToList();
143  if (domain.GatewaySettings != null)
144  {
145  domain.GatewaySettings.Groups = groupsWithSetId
146  .Where(x => x.s == domain.GroupSetId)
147  .Select(x => x.g).ToList();
148  }
149  }
