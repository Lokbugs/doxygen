104  public void ValidateTimetable(ForwardingSettings forwardingSettings)
105  {
106  var forwardingsToNumber = _forwardingRepository.GetByUserId(forwardingSettings.AbonentId).ToList();
107 
108  // Выберем идентификаторы всех правил переадресации указанного типа для абонента, кроме самого проверяемого правила.
109  var matchingAdditionalForwardingIds = forwardingsToNumber.Where(x => x.State == forwardingSettings.State && x.Id.HasValue &&
110  (!forwardingSettings.Id.HasValue || forwardingSettings.Id.Value != x.Id.Value)).Select(x => x.Id.Value).ToList();
111 
112  var timeTableIds = _forwardingRepository.GetTimetableIdsByAdditionalForwardingIds(matchingAdditionalForwardingIds.Distinct()).ToList();
113 
114  var newTimetable = new (string DayName, DayOfWeekSchedule Schedule)[]
115  {
116  ("monday", forwardingSettings.Timetable.Monday),
117  ("tuesday", forwardingSettings.Timetable.Tuesday),
118  ("wednesday", forwardingSettings.Timetable.Wednesday),
119  ("thursday", forwardingSettings.Timetable.Thursday),
120  ("friday", forwardingSettings.Timetable.Friday),
121  ("saturday", forwardingSettings.Timetable.Saturday),
122  ("sunday", forwardingSettings.Timetable.Sunday)
123  };
124  _baseRepository.GetBy<rtu.timetable_week_day>(new { timetable = timeTableIds.ToList() }, new
125  {
126  timetable = (int)0,
127  week_day_name = (string)null,
128  interval_begin = (TimeSpan?)TimeSpan.Zero,
129  interval_end = (TimeSpan?)TimeSpan.Zero
130  }.GetType())
131  .Select(x => ((int)x.timetable, (string)x.week_day_name, (TimeSpan?)x.interval_begin, (TimeSpan?)x.interval_end))
132  .GroupBy(x => x.Item1)
133  .Select(x => x.Select(f => (f.Item2, f.Item3, f.Item4)).ToList())
134  .ToList()
135  .ForEach(x => _timetableValidationService.ValidateTimetableIntersect(newTimetable, x));
136  }

143  public void ValidateDestinationType(ForwardingSettings forwardingSettings)
144  {
145  Regex regex = new Regex(COMMON_PHONE_REGEXP);
146 
147  var lineTypeCode = _lineTypeService.GetLineTypeCodeByAbonentId(forwardingSettings.AbonentId);
148 
149  if (!lineTypeCode.IsNullOrEmpty())
150  {
151  if (lineTypeCode.Equals(EXTERNAL_NUMBER_LINE_TYPE))
152  throw new InvalidNumberTypeException();
153 
154  if (!regex.IsMatch(forwardingSettings.ForwardNumber) || (forwardingSettings.DestinationType!= null && forwardingSettings.DestinationType != ForwardingDestinationType.ExternalNumber))
155  throw new ExternalNumberIncompliantException();
156  }
157 
158  if (forwardingSettings.DestinationType == ForwardingDestinationType.InternalNumber
159  && !_vtePortRepository.CheckInternalNumberDestinationTypeByLocalNumber(forwardingSettings.AbonentId, forwardingSettings.ForwardNumber))
160  throw new InternalNumberIncompliantException();
161 
162  var domainIds = _baseRepository.GetBy<rtu.user>(new {user = forwardingSettings.AbonentId}, new {domain = (int) 0}.GetType()).Select(x => (int)x.domain).ToList();
163  if (forwardingSettings.DestinationType == ForwardingDestinationType.Group
164  && (!domainIds.Any() || !_baseRepository.GetBy<rtu.user_group>(new { domain = domainIds, number = Int32.Parse(forwardingSettings.ForwardNumber)}, new { user_group = (int)0}.GetType()).Any()))
165  throw new GroupIncompliantException();
166  }

232  public void ValidateCallDirectionAvailability(int userId, string phoneForward, DateTime dt)
233  {
234  var callDirectionInfo = _callDirectionService.GetCallDirectionInfo(userId, phoneForward, dt);
235 
236  // Во всех неоднозначных случаях, когда нет возможности определить направление и настроки,

245  throw new CallDirectionNotAllowedException();
246  break;
247  case CallDirection.mg:
248  if(!callDirectionInfo.FarContract.HasValue)
249  throw new CallDirectionNotAllowedException();
250  break;
251  case CallDirection.city:
252  case CallDirection.zone:

254  throw new CallDirectionNotAllowedException();
