194  public void Update(ForwardingSettings forwarding_settings, bool notify_phone_config = true)
195  {
196  if (!forwarding_settings.Id.HasValue)

198  throw new ForwardingIdIsNullException();

200  _forwardingValidationService.ValidateDestinationType(forwarding_settings);

206  var isStateBusy = forwarding_settings.State == State.Busy;
