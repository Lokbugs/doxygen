◆ _getChangingNumbersByDomainId

readonly string Cordis.Core.Sip.Class5.Settings.Vtes.Repository.VteRepository._getChangingNumbersByDomainId
private
Initial value:
= @"
select vvp.local_number
from RTU.domain d
join RTU.[user] u on u.domain = d.domain
join Voice.vte_port vvp on vvp.rtu_user = u.[user]
where len(vvp.local_number) = 3
and d.domain = @domainId
union
select rug.number
from RTU.user_group rug
where len(rug.number) = 3
and rug.domain = @domainId
union
select cast(rs.number as varchar(24))
from RTU.service rs
where len(rs.number) = 3
and rs.domain = @domainId
"
