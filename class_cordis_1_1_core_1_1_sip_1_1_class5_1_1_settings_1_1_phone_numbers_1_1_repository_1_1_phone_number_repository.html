readonly string Cordis.Core.Sip.Class5.Settings.PhoneNumbers.Repository.PhoneNumberRepository.GetPhoneNumberByUserIdsQuery
private
Initial value:
= @"
select
vp.rtu_user,
pn.phone_number
from Voice.vte_port vp
join rtu.[user] u on u.[user] = vp.rtu_user
join rtu.gateway g on g.domain = u.domain
join RTU.gateway__translation gt on gt.src_match = '^' + vp.local_number + '$' and gt.gateway = g.gateway
join Voice.phone_list pl on gt.result = '7' + cast(pl.prefix as varchar(max)) + cast(pl.name as varchar(max))
join Voice.phone_number pn on pn.first_index = pl.name
join Sim.phone_number s on s.phone_number = pn.phone_number
join Configuration.getSimBinding(getdate()) sb on sb.sim = s.sim
join @ids ids on ids.ID = vp.rtu_user
"

Запрос для получения идентификаторов телефонных номеров по идентификаторам портов ВАТС в таблице RTU.user.
